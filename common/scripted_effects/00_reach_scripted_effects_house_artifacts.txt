		type = sword
		wealth = scope:wealth
		quality = scope:quality
		template = baseborn_redwyne_artifact_template

			type = created

		save_scope_as = owner
	}
	set_artifact_rarity_famed = yes
	create_artifact = {
		name = baseborn_meadows_artifact_name
		creator = character:Meadows_67 # Remove this line from the code if the creator is unknown
		description = baseborn_meadows_artifact_description
		visuals = longsword
		type = sword
		wealth = scope:wealth
		quality = scope:quality
		template = baseborn_meadows_artifact_template # Remove this line from the code if the artifact has no restrictions
		history = {
			type = created
			date = 7966.1.1 # Date the artifact was created
			recipient = character:Meadows_67 # First character to use the artifact
			location = province:4041 # Location of the first character to use the artifact
		}
		modifier = baseborn_meadows_artifact_modifier
		save_scope_as = newly_created_artifact
		decaying = yes
	}
	scope:newly_created_artifact ?= {
		equip_artifact_to_owner_replace = yes
		set_variable = {
			name = historical_unique_artifact
			value = yes
		}
		set_variable = {
			name = traditional_house_owner
			value = dynasty:dynn_Meadows.dynasty_founder.house
		}

		agot_add_artifact_history = {
			ARTIFACT = this
			TYPE = inherited
			DATE = 8015.1.1
			RECIPIENT = character:Meadows_71 # Next user of the artfact
		}

		agot_add_artifact_history = {
			ARTIFACT = this
			TYPE = inherited
			DATE = 8021.1.1
			RECIPIENT = character:Meadows_13 # Next user of the artfact
		}
		
		agot_add_artifact_history = {
			ARTIFACT = this
			TYPE = inherited
			DATE = 8042.1.1
			RECIPIENT = character:Meadows_14 # Next user of the artfact
		}

		agot_add_artifact_history = {
