baseborn_glover_artifact_template = {
	can_equip = {
		always = yes
	}
	# can this character benefit from the full modifiers of the artifact?
	can_benefit = {
		dynasty = dynasty:dynn_Glover
	}
	# if a given character does not pass the "can_benefit" trigger then this modifier will be applied instead.
	fallback = {
        prowess_no_portrait = 3 # change this number to the amount of prowess given by the weapon
        general_opinion = -15
	}
	unique = yes
	ai_score = {
        value = 0
        if = {
            limit = {
                dynasty ?= dynasty:dynn_Glover
            }
            add = 100
        }
	}
}
