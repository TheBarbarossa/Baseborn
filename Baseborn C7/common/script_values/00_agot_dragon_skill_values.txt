	value = 0

		}

wild_dragons_in_realm = {
	value = 0
	if = {
		limit = {
			any_courtier = {
				has_trait = dragon
			}
		}
		every_courtier = {
			limit = {
				AND = {
					has_trait = dragon
					NOR = {
						has_character_flag = owned_dragon
						has_character_flag = in_dragonpit
						has_variable = current_rider
					}
				}
			}
			add = 1
		}
	}
}
